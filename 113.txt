 #include <iostream>

using namespace std;

 class Food{
public:
    void SetPrice(int myprice){
        price = myprice;
    }
    int GetPrice(){
        return price;
    }
private:
    int price;
};

class Fruit : public Food{
public:
    void SetPrice(int myprice){
        Food::SetPrice(myprice - 20);
    }
};

int main()
{
    Food myFood;
    Fruit myFruit;

    myFood.SetPrice(100);
    myFruit.SetPrice(100);

    cout << "myFoodªºprice = " << myFood.GetPrice() << endl;
    cout << "myFruitªºprice = " << myFruit.GetPrice() << endl;
    return 0;
}
